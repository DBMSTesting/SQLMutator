<cursor properties> ::=
[ <cursor sensitivity> ] [ <cursor scrollability> ] CURSOR
 [ <cursor holdability> ]
 [ <cursor returnability> ]
<cursor sensitivity> ::=
SENSITIVE
| INSENSITIVE
| ASENSITIVE
<cursor scrollability> ::=
SCROLL
| NO SCROLL
<cursor holdability> ::=
WITH HOLD
| WITHOUT HOLD
<cursor returnability> ::=
WITH RETURN
| WITHOUT RETURN
