
CREATE TABLE p (
  id INTEGER PRIMARY KEY,
  uid VARCHAR(36),
  t INTEGER
);

CREATE TABLE pa (
  id INTEGER PRIMARY KEY,
  a_uid VARCHAR(36)
);

CREATE TABLE a (
  id INTEGER PRIMARY KEY,
  uid VARCHAR(36),
  t INTEGER
);

SELECT DISTINCT *
FROM 
  pa
  LEFT JOIN p ON p.uid='1234'
  LEFT JOIN a ON a.uid=pa.a_uid
WHERE 
  a.t=p.t;
