CREATE TABLE a(id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, a TEXT);
CREATE VIEW view_a AS SELECT * FROM a;
CREATE TRIGGER view_a_insert INSTEAD OF INSERT ON view_a
BEGIN
  INSERT INTO a (a) VALUES (NEW.a);
END;
