Title: DuckDB lag seems to be much slower than Polars
Description:
### What happens?

As reported in https://github.com/ibis-project/ibis/issues/9405, the lag operation in Polars appears to be much faster than in DuckDB.

### To Reproduce

```python
import time
import duckdb
import polars as pl
import numpy as np


rng = np.random.default_rng(1)
N = 10_000_000
a = rng.normal(size=N)
b = rng.normal(size=N)
c = rng.normal(size=N)
data = {'a': a, 'b': b, 'c': c}
df_pl = pl.LazyFrame(data)

cols = ["a", "b", "c"]
lags = [1, 2]


def add_lags_polars():
    return df_pl.select(
        pl.col(col).shift(-lag).alias(f'{col}_{lag}')
        for col in cols
        for lag in lags
    )


def add_lags_duckdb():
    sql = """
SELECT
  LAG(a, -1) OVER () AS a_1,
  LAG(a, -2) OVER () AS a_2,
  LAG(b, -1) OVER () AS b_1,
  LAG(b, -2) OVER () AS b_2
FROM df
    """
    rel = con.sql(sql)
    return rel.pl()

start = time.time()
add_lags_polars().collect()
stop = time.time()
print(f"polars:   {stop - start:.4f}s")

con = duckdb.connect()
con.execute("CREATE OR REPLACE TABLE df AS FROM df_pl")
start = time.time()
add_lags_duckdb()
stop = time.time()
print(f"duckdb:   {stop - start:.4f}s")
```
